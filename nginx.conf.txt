server {
    listen 80;
    
    root /usr/share/nginx/html;
    index index.html index.htm;

    # Frontend (React)
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Proxy Seguro para Evolution API (WhatsApp Gateway)
    # Redireciona /evolution/xyz para http://whatsapp-gateway:8080/xyz
    location /evolution/ {
        proxy_pass http://whatsapp-gateway:8080/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # Proxy Seguro para SalesBot API (Backend)
    # Redireciona /api/xyz para http://salesbot-api:8080/api/xyz
    location /api/ {
        proxy_pass http://salesbot-api:8080/api/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /usr/share/nginx/html;
    }
    
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, no-transform";
    }
}
