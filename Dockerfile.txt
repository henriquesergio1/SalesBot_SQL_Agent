# Estágio 1: Build da Aplicação React
FROM node:18-alpine as build

WORKDIR /app

# Copia arquivos de dependência
COPY package*.json ./
RUN npm install

# Copia código fonte e gera o build de produção
COPY . .
RUN npm run build

# Estágio 2: Servidor Web Nginx
FROM nginx:alpine

# Copia o build gerado para o diretório do Nginx
COPY --from=build /app/dist /usr/share/nginx/html

# Copia a configuração customizada do Nginx (renomeando para .conf)
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Expõe a porta 80
EXPOSE 80

# Inicia o Nginx
CMD ["nginx", "-g", "daemon off;"]
