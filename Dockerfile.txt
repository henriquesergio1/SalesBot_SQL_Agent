# Estágio 1: Build da Aplicação React
FROM node:20-alpine as build

WORKDIR /app

# Copia apenas o package.json (Ignora package-lock.json para evitar conflitos de OS)
COPY package.json ./

# Instala dependências sem ler o lockfile (--no-package-lock) e sem auditoria
# Isso força o download dos binários corretos para Linux (Alpine)
RUN npm install --legacy-peer-deps --no-audit --no-package-lock

# Copia o restante do código fonte
COPY . .

# Gera o build de produção
RUN npm run build

# Estágio 2: Servidor Web Nginx
FROM nginx:alpine

# Remove configurações padrão do Nginx
RUN rm -rf /etc/nginx/conf.d/*

# Copia o build gerado no estágio anterior
COPY --from=build /app/dist /usr/share/nginx/html

# Copia a configuração customizada do Nginx
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Expõe a porta 80
EXPOSE 80

# Inicia o Nginx
CMD ["nginx", "-g", "daemon off;"]
