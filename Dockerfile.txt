# Estágio 1: Build da Aplicação React
# Usando a imagem completa (bullseye) para garantir que temos Python/Make/GCC para o build do Vite
FROM node:18 as build

WORKDIR /app

# Copia arquivos de configuração
COPY package*.json ./

# Instala dependências
# --legacy-peer-deps: Evita conflitos de versões
# --no-audit: Acelera a instalação ignorando auditoria de segurança
RUN npm install --legacy-peer-deps --no-audit

# Copia todo o código fonte
COPY . .

# Gera o build de produção
RUN npm run build

# Estágio 2: Servidor Web Nginx
FROM nginx:alpine

# Remove configurações padrão do Nginx
RUN rm -rf /etc/nginx/conf.d/*

# Copia o build gerado no estágio anterior
COPY --from=build /app/dist /usr/share/nginx/html

# Copia a configuração customizada do Nginx
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Expõe a porta 80
EXPOSE 80

# Inicia o Nginx
CMD ["nginx", "-g", "daemon off;"]
